26 // https://stackoverflow.com/questions/24518989/how-to-perform-1-dimensional-valid-convolution
27 // The conv function implements the standard convolution method,
28 // where the output size is the sum of input sizes minus one.
29 template<typename T>
30 std::vector<T> conv(std::vector<T> const& f, std::vector<T> const& g)
31 {
32 int const nf = f.size();
33 int const ng = g.size();
34 int const n = nf + ng - 1;
35 std::vector<T> out(n, T());
36 for (auto i(0); i < n; ++i) {
37 int const jmn = (i >= ng - 1) ? i - (ng - 1) : 0;
38 int const jmx = (i < nf - 1) ? i : nf - 1;
39 for (auto j(jmn); j <= jmx; ++j) {
40 out[i] += (f[j] * g[i - j]);
41 }
42 }
43 return out;
44 }

47 // which is generally more efficient for some applications where zero-padded areas

50 // https://stackoverflow.com/questions/24518989/how-to-perform-1-dimensional-valid-convolution
51 // The conv_valid function implements the valid convolution method,
52 // where the output size is the absolute difference between the input sizes plus one.
53 template<typename T>
54 std::vector<T> conv_valid(std::vector<T> const& f, std::vector<T> const& g)
55 {
56 int const nf = f.size();
57 int const ng = g.size();
58 std::vector<T> const& min_v = (nf < ng) ? f : g;
59 std::vector<T> const& max_v = (nf < ng) ? g : f;
60 int const n = std::max(nf, ng) - std::min(nf, ng) + 1;
61 std::vector<T> out(n, T());
62 for (auto i(0); i < n; ++i) {
63 for (int j(min_v.size() - 1), k(i); j >= 0; --j) {
64 out[i] += min_v[j] * max_v[k];
65 ++k;
66 }
67 }
68 return out;
69 }

73 std::vector<T> conv_same(std::vector<T> const& f, std::vector<T> const& g) {
74 int const nf = f.size();
75 int const ng = g.size();
76 std::vector<T> out(nf, T()); // Output size matches the size of f for 'same' mode
77
78 // Calculate padding needed for 'same' mode, adjusted for shift correction
79 const int pad = (ng - 1) / 2;
80
81 for (int i = 0; i < nf; ++i) {
82 T sum = T(); // Initialize sum for each element of output
83 for (int j = 0; j < ng; ++j) {
84 int fIndex = i + j - pad; // Adjust index for 'same' mode with shift correction
85 if (fIndex >= 0 && fIndex < nf) { // Check boundaries
86 sum += f[fIndex] * g[j];
87 }
88 }
89 out[i] = sum;
90 }
91 return out;
92 }

152 template<typename T>
153 std::vector<T> circularGauss(uint32_t size, const T& sigma, const T& position)
154 {
155 uint32_t l = size - 2 * 1 + 2;
156 uint32_t m = 1;
157 T r = position - static_cast<T>(m); // Calculate the shift with fractional part
158 T rem = std::fmod(r, static_cast<T>(l)); // Calculate the remainder
159 T positionShifted = rem + static_cast<T>(m); // Apply the shifted position
160
161 std::vector<T> g(size);
162 std::vector<T> xRange(size);
163 std::iota(xRange.begin(), xRange.end(), static_cast<T>(1));
164
165 std::vector<T> d(size);
166 std::vector<T> lMinusd(size);
167 std::transform(xRange.begin(), xRange.end(), d.begin(), [&positionShifted, &l](T element) { return std::abs(element - positionShifted); });
168 std::transform(d.begin(), d.end(), lMinusd.begin(), [&l](T element) { return -1 * (element - l); });
169 for (int i = 0; i < size; i++)
170 {
171 g[i] = std::exp(-0.5 * std::pow(std::min(d[i], lMinusd[i]), 2) / std::pow(sigma, 2));
172 }
173
174 return g;
175 }

178 std::vector<T> gaussDerivative(const std::vector<int>& rangeX, const T& position, const T& sigma, const T& amplitude)
179 {
180 std::vector<T> derivative(rangeX.size());
181 T variance = sigma * sigma;
182
183 for (size_t i = 0; i < derivative.size(); i++)
184 {
185 T x = rangeX[i];
186 derivative[i] = -(x - position) / variance * amplitude * std::exp(-0.5 * std::pow((x - position), 2) / variance);
187 }
188
189 return derivative;

254 {
255 static constexpr T epsilon = 1e-6;
256 static constexpr T offset = 1.0;
257
258 // Find the minimum and maximum values in the vector
259 const T minVal = *std::ranges::min_element(vector.begin(), vector.end()) - epsilon;
260
261 std::vector<T> normalizedVector = vector;
262 for (T& val : normalizedVector)
263 val += minVal;

266 const T newMinVal = *std::ranges::min_element(normalizedVector.begin(), normalizedVector.end());

268 for (T& val : normalizedVector)
269 val -= newMinVal;
270
271 return normalizedVector;

275 std::vector<T> hebbLearningRule(std::vector<T>& weights, const std::vector<T>& input, const std::vector<T>& output, double learningRate)
276 {
277 if (input.empty() || output.empty())
278 throw std::invalid_argument("Input and output vectors cannot be empty");
279
280 const size_t inputSize = input.size();
281 const size_t outputSize = output.size();
282 const size_t expectedSize = inputSize * outputSize;
283
284 if (weights.size() != expectedSize)
285 throw std::invalid_argument("Weight matrix size mismatch");
286
287 for (size_t i = 0; i < inputSize; ++i)
288 {
289 const T scaledInput = learningRate * input[i];
290 const size_t baseIndex = i * outputSize;
291
292 for (size_t j = 0; j < outputSize; ++j)
293 weights[baseIndex + j] += scaledInput * output[j];
294 }
295
296 return weights;
297 }

457 // Linear interpolation resampling from input.size() to outputSize.
458 template<typename T>
459 std::vector<T> resample(const std::vector<T>& input, int outputSize)
460 {
461 if (input.empty() || outputSize <= 0) return {};
462 const int N = static_cast<int>(input.size());
463 if (N == outputSize) return input;
464 if (outputSize == 1) return { input[N / 2] };
465 std::vector<T> out(outputSize);
466 for (int i = 0; i < outputSize; ++i)
467 {
468 const double pos = static_cast<double>(i) * (N - 1) / (outputSize - 1);
469 const int lo = static_cast<int>(pos);
470 const int hi = std::min(lo + 1, N - 1);
471 const double t = pos - lo;
472 out[i] = input[lo] * (1.0 - t) + input[hi] * t;
473 }
474 return out;
475 }

478 template<typename T>

480 {

482 const int outputSize = static_cast<int>(output.size());
483 if (input.empty() || outputSize <= 0) return;

491 const double t = pos - lo;
